-- Tạo bảng tasks để thực hành CRUD
CREATE TABLE tasks (
  id          SERIAL PRIMARY KEY,
  title       VARCHAR(200) NOT NULL,
  description TEXT,
  status      VARCHAR(20) DEFAULT 'pending'
                CHECK (status IN ('pending', 'in_progress', 'done')),
  priority    INTEGER DEFAULT 5 CHECK (priority BETWEEN 1 AND 10),
  created_at  TIMESTAMPTZ DEFAULT NOW(),
  updated_at  TIMESTAMPTZ DEFAULT NOW()
);

-- Seed data test
INSERT INTO tasks (title, status, priority) VALUES
  ('Setup n8n workflows', 'done', 9),
  ('Integrate Telegram bot', 'in_progress', 7),
  ('Write documentation', 'pending', 5);
4 Workflow CRUD cơ bản
📥
SELECT — Lấy danh sách
Webhook GET → PostgreSQL Execute → Respond to Webhook với JSON array tasks.
📤
INSERT — Tạo mới
Webhook POST → Validate data → PostgreSQL Insert → Return created item với id.
✏️
UPDATE — Cập nhật
Webhook PUT → Check tồn tại → PostgreSQL Update → Return updated item.
🗑️
DELETE — Xóa
Webhook DELETE → Soft delete (update status) → Return success message.
Query mẫu cho PostgreSQL node
PostgreSQL Node — Parameterized QueriesSQL
-- SELECT với filter động (dùng Expression mode)
SELECT * FROM tasks
WHERE status = $1
ORDER BY priority DESC, created_at DESC;
-- Query Params: {{ $json.status }}

-- INSERT với return id
INSERT INTO tasks (title, description, priority)
VALUES ($1, $2, $3)
RETURNING *;
-- Query Params: $json.title, $json.description, $json.priority

-- UPDATE
UPDATE tasks
SET status = $1, updated_at = NOW()
WHERE id = $2
RETURNING *;
-- Query Params: $json.status, $json.id

// Tool Name: calculator
// Description: "Use to evaluate mathematical expressions. Input: math expression string"

const expression = $input.first().json.query;

try {
  // Dùng Function để eval an toàn
  const allowed = /^[\d\s\+\-\*\/\(\)\.\,\%\^]+$/;
  if (!allowed.test(expression)) throw new Error('Invalid expression');

  const result = new Function(`"use strict"; return (${expression})`)();
  return [{ json: { result: result, expression: expression } }];
} catch(e) {
  return [{ json: { error: `Cannot evaluate: ${expression}` } }];
}

Backup tự động hàng ngày
backup-n8n.sh — Chạy bằng cron 2:00 AMBash
#!/bin/bash
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backups/n8n"
KEEP_DAYS=30

mkdir -p "$BACKUP_DIR"

# Backup PostgreSQL
docker exec n8n-postgres pg_dump -U n8n n8n \
  | gzip > "$BACKUP_DIR/db_$DATE.sql.gz"

# Backup n8n data volume
docker run --rm \
  -v n8n_data:/data \
  -v "$BACKUP_DIR":/backup \
  alpine tar czf /backup/n8n_data_"$DATE".tar.gz /data

# Xóa backup cũ hơn 30 ngày
find "$BACKUP_DIR" -name "*.gz" -mtime +"$KEEP_DAYS" -delete

echo "✅ Backup completed: $DATE"
