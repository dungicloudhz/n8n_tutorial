Toán tử (Operators)

Tính toán và so sánh giá trị

+ - * /
Số học cơ bản
5 + 3 → 8
%
Chia lấy dư
10 % 3 → 1
**
Lũy thừa
2 ** 3 → 8
===
Bằng (nghiêm ngặt)
5 === '5' → false
!==
Khác nhau
5 !== 6 → true
&& || !
Logic AND/OR/NOT
true && false → false
operators.js 
// Số học
console.log('10 + 3 =', 10 + 3);
console.log('10 % 3 =', 10 % 3);    // chia lấy dư
console.log('2 ** 8 =', 2 ** 8);    // lũy thừa

// So sánh — luôn dùng === thay ==
console.log('5 == "5"  :', 5 == '5');     // true  (nguy hiểm!)
console.log('5 === "5" :', 5 === '5');    // false (an toàn ✅)

// Logic
console.log(true && false);   // AND
console.log(true || false);   // OR
console.log(!true);            // NOT

// Toán tử 3 ngôi
const diem = 75;
const xepLoai = diem >= 50 ? 'Đậu ✅' : 'Rớt ❌';
console.log('Kết quả:', xepLoai);
Output
⚠️Luôn dùng === thay vì ==. Toán tử == tự ép kiểu có thể gây bug khó tìm!

Mảng (Array)

Lưu danh sách các giá trị

arrays.js 
const fruits = ['xoài', 'cam', 'táo', 'chuối'];

// Truy cập (index bắt đầu từ 0)
console.log(fruits[0]);             // 'xoài'
console.log(fruits.length);          // 4

// Thêm / xóa
fruits.push('nho');
console.log('Sau push:', fruits);

const cuoi = fruits.pop();
console.log('Đã pop:', cuoi, '| còn:', fruits);

// map — tạo mảng mới
const viethoa = fruits.map(f => f.toUpperCase());
console.log('map:', viethoa);

// filter — lọc theo điều kiện
const so = [1,2,3,4,5,6];
console.log('Số chẵn:', so.filter(n => n % 2 === 0));

// reduce — tính tổng
const tong = so.reduce((acc, n) => acc + n, 0);
console.log('Tổng 1→6:', tong);

// includes & find
console.log('Có cam?', fruits.includes('cam'));
console.log('find >3:', so.find(n => n > 3));

Object (Đối tượng)

Nhóm các thuộc tính liên quan lại

objects.js 
const nguoi = {
  ten: 'Minh',
  tuoi: 25,
  thanhPho: 'Hà Nội',
  hobbies: ['code', 'đọc sách'],
  chaoHoi() {
    return `Tôi là ${this.ten}, ${this.tuoi} tuổi`;
  }
};

// Truy cập
console.log(nguoi.ten);
console.log(nguoi['tuoi']);
console.log(nguoi.chaoHoi());

// Thêm / sửa / xóa
nguoi.email = 'minh@email.com';
nguoi.tuoi  = 26;
delete nguoi.thanhPho;
console.log(nguoi);

// Destructuring — gán nhanh
const { ten, tuoi } = nguoi;
console.log('Destructure:', ten, tuoi);

// Spread — sao chép / gộp
const ban = { ...nguoi, ten: 'Nam' };
console.log('Spread copy:', ban.ten, ban.tuoi);
Output

Điều kiện if / else

Rẽ nhánh logic theo điều kiện

conditions.js 
const diem = 75; // Thử đổi giá trị này!

// if / else if / else
if (diem >= 90) {
  console.log('🏆 Xuất sắc');
} else if (diem >= 70) {
  console.log('✅ Khá');
} else if (diem >= 50) {
  console.log('📝 Trung bình');
} else {
  console.log('❌ Yếu');
}

// Switch — nhiều trường hợp cụ thể
const thu = 'Thứ Hai'; // Thử đổi!
switch (thu) {
  case 'Thứ Hai': console.log('Đầu tuần 😩'); break;
  case 'Thứ Sáu': console.log('Sắp cuối tuần 🎉'); break;
  case 'Chủ Nhật': console.log('Nghỉ ngơi 😴'); break;
  default: console.log('Ngày thường 💼');
}

// Optional chaining ?. — tránh lỗi null
const user = { profile: { name: 'Minh' } };
console.log(user?.profile?.name);    // 'Minh'
console.log(user?.address?.city);    // undefined (không lỗi)

Vòng lặp (Loops)

Lặp đi lặp lại các tác vụ

loops.js 
// for — biết trước số lần lặp
for (let i = 1; i <= 5; i++) {
  console.log(`Lần ${i}: ${'⭐'.repeat(i)}`);
}

// for...of — duyệt mảng (hiện đại ✅)
const fruits = ['🥭 xoài', '🍊 cam', '🍎 táo'];
for (const f of fruits) {
  console.log(f);
}

// for...in — duyệt object
const nguoi = { ten: 'Minh', tuoi: 25, tp: 'HN' };
for (const key in nguoi) {
  console.log(`${key} → ${nguoi[key]}`);
}

// break & continue
console.log('--- break/continue ---');
for (let i = 1; i <= 8; i++) {
  if (i === 4) { console.log('Bỏ qua 4'); continue; }
  if (i === 7) { console.log('Dừng tại 7'); break; }
  console.log(i);
}

Hàm (Functions)

Đóng gói và tái sử dụng code

functions.js 
// 1. Khai báo hàm thông thường
function chaoHoi(ten = 'bạn') {
  return `Xin chào, ${ten}! 👋`;
}
console.log(chaoHoi('Minh'));
console.log(chaoHoi());           // dùng giá trị mặc định

// 2. Arrow Function — cú pháp hiện đại ✅
const nhan = (a, b) => a * b;    // return ngầm
const luyThua = (a, b) => {
  const kq = a ** b;
  return `${a}^${b} = ${kq}`;
};
console.log(nhan(6, 7));
console.log(luyThua(2, 10));

// 3. Rest params — nhận N tham số
const tong = (...nums) =>
  nums.reduce((a, b) => a + b, 0);
console.log('Tổng:', tong(1,2,3,4,5));

// 4. Higher-order function — hàm nhận hàm
const apDung2Lan = (fn, x) => fn(fn(x));
console.log(apDung2Lan(x => x * 2, 3));  // 3→6→12
Output

DOM — Tương tác với HTML

Điều khiển trang web bằng JavaScript

DOM (Document Object Model) là cách JS "nhìn thấy" trang HTML. Mỗi thẻ HTML là một node có thể đọc và thay đổi.

dom.js 
// Tạo và thêm phần tử vào DOM demo bên dưới
const box = document.getElementById('dom-demo');
box.innerHTML = ''; // Xóa nội dung cũ

// Tạo tiêu đề
const h = document.createElement('h4');
h.textContent = '📦 Phần tử được tạo bởi JS!';
h.style.color = '#f5c842';
box.appendChild(h);

// Tạo danh sách
const fruits = ['🥭 Xoài', '🍊 Cam', '🍎 Táo'];
const ul = document.createElement('ul');
ul.style.cssText = 'list-style:none;padding:0;margin:10px 0';

fruits.forEach((fruit, i) => {
  const li = document.createElement('li');
  li.textContent = fruit;
  li.style.cssText = `padding:6px 10px;margin:4px 0;
    background:rgba(78,205,196,0.1);border-radius:5px;cursor:pointer;
    color:#4ecdc4;animation:fadeIn .3s ${i*0.1}s both`;
  li.onclick = () => li.style.textDecoration =
    li.style.textDecoration ? '' : 'line-through';
  ul.appendChild(li);
});

box.appendChild(ul);
console.log('DOM đã được cập nhật! Click vào items bên dưới 👇');

Sự kiện (Events)

Phản ứng với hành động của người dùng

events.js 
// Gắn sự kiện vào các nút demo bên dưới
const btnClick = document.getElementById('ev-click');
const inp      = document.getElementById('ev-input');
const evLog    = document.getElementById('ev-log');
let dem = 0;

const log = (msg) => {
  evLog.innerHTML = `<span style="color:var(--green)">${msg}</span>`;
};

// click event
btnClick.addEventListener('click', () => {
  dem++;
  log(`🖱️ Click số ${dem}!`);
  console.log('Click!', dem);
});

// input event — gõ từng ký tự
inp.addEventListener('input', (e) => {
  log(`⌨️ Đang gõ: "${e.target.value}" (${e.target.value.length} ký tự)`);
  console.log('input:', e.target.value);
});

// keydown — phím nào được nhấn
inp.addEventListener('keydown', (e) => {
  if (e.key === 'Enter') log('⏎ Bạn nhấn Enter!');
});

console.log('✅ Events đã được gắn! Thử tương tác bên dưới 👇');
Output

Bất đồng bộ (Async/Await)

Gọi API và xử lý tác vụ mất thời gian

Một số tác vụ cần thời gian như gọi API. async/await giúp "chờ" kết quả mà không làm đơ trang.

async.js 
// Gọi API thật — JSONPlaceholder (API test miễn phí)
async function layUser(id) {
  console.log(`⏳ Đang lấy user #${id}...`);
  try {
    const res  = await fetch(
      `https://jsonplaceholder.typicode.com/users/${id}`
    );
    const user = await res.json();
    console.log('✅ Tên:',   user.name);
    console.log('📧 Email:', user.email);
    console.log('🏙️ Thành phố:', user.address.city);
    return user;
  } catch (err) {
    console.error('❌ Lỗi:', err.message);
  }
}

// Gọi hàm — thử đổi id từ 1-10
layUser(1);

// setTimeout demo
console.log('⚡ Dòng này chạy NGAY, không chờ API');
setTimeout(() => console.log('⏰ Chạy sau 1 giây'), 1000);
Output
✅Ví dụ này gọi API thật từ jsonplaceholder.typicode.com! Thử đổi số 1 thành 2, 3... để lấy user khác.

Ví dụ thực tế — To-do List

Kết hợp tất cả kiến thức vào app mini

todo.js 
// Khởi tạo To-do List hoàn chỉnh
const todoInput = document.getElementById('todo-inp');
const todoList  = document.getElementById('todo-list');
const todoBtn   = document.getElementById('todo-btn');
const todoCount = document.getElementById('todo-count');

let todos = [];

const render = () => {
  todoList.innerHTML = todos.map((t, i) => `
    <li style="display:flex;justify-content:space-between;
      align-items:center;padding:9px 12px;margin:5px 0;
      background:var(--surface3);border-radius:7px;
      border-left:3px solid ${t.done ? 'var(--green)' : 'var(--js)'}">
      <span onclick="toggleTodo(${i})" style="cursor:pointer;flex:1;
        text-decoration:${t.done ? 'line-through' : 'none'};
        opacity:${t.done ? '.45' : '1'};color:var(--text);font-size:13px">
        ${t.done ? '✅' : '⬜'} ${t.text}
      </span>
      <button onclick="removeTodo(${i})" style="background:var(--red-dim);
        border:none;color:var(--red);padding:3px 8px;border-radius:4px;
        cursor:pointer;font-size:11px;font-weight:700">✕</button>
    </li>`).join('');
  todoCount.textContent =
    `${todos.filter(t=>t.done).length}/${todos.length} hoàn thành`;
};

window.toggleTodo = (i) => {
  todos[i].done = !todos[i].done; render();
};
window.removeTodo = (i) => {
  todos.splice(i, 1); render();
};

const addTodo = () => {
  const txt = todoInput.value.trim();
  if (!txt) return;
  todos.push({ text: txt, done: false });
  todoInput.value = '';
  render();
  console.log('Thêm:', txt);
};

todoBtn.addEventListener('click', addTodo);
todoInput.addEventListener('keydown', e =>
  e.key === 'Enter' && addTodo());

// Thêm vài task mẫu
['Học HTML & CSS', 'Học JavaScript', 'Làm dự án đầu tiên']
  .forEach(t => todos.push({ text: t, done: false }));
todos[0].done = true;
render();
console.log('✅ To-do List đã sẵn sàng!');
Output
